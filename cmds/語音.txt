every_min0_time = [
    datetime.time(hour = i, minute = 0, tzinfo = datetime.timezone(datetime.timedelta(hours = 8)))
    for i in range(24)
]

class fire0(commands.Cog):
    tz = datetime.timezone(datetime.timedelta(hours = 8))

    def __init__(self, bot: commands.Bot):
        self.bot = bot
        self.everyday_0.start()
    
    @tasks.loop(time=every_min0_time)
    async def everyday_0(self):
        channel_id = 1353676679800946698
        channel = self.bot.get_channel(channel_id)
        
        # Connect to the voice channel
        if not channel.guild.voice_client:
            await channel.connect()
        
        voice_client = channel.guild.voice_client
        
        # Check if voice client is playing or paused
        if voice_client.is_playing() or voice_client.is_paused():
            return
        
        text = "記得補火"
        tts = gTTS(text, lang='zh')
        filename = 'tts.mp3'
        tts.save(filename)
        
        # Play the audio
        voice_client.play(discord.FFmpegPCMAudio(source=filename), after=lambda e: os.remove(filename))

    def cog_unload(self):
        self.everyday_0.cancel()

async def setup(bot: commands.Bot):
    await bot.add_cog(fire0(bot))
